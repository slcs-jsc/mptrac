46  char *argv[]) {

48  ctl_t ctl;

50  met_t *met;

54  static double p2[1000], t[1000], t2[1000], z[1000], z2[1000], lat_mean,
55  z_mean;
56 
57  static int hist_max[1000], hist_min[1000], hist_mean[1000], hist_sig[1000],
58  nhist_max, nhist_min, nhist_mean, nhist_sig, np;
59 
60  /* Allocate... */
61  ALLOC(met, met_t, 1);
62 
63  /* Check arguments... */
64  if (argc < 4)
65  ERRMSG("Give parameters: <ctl> <hist.tab> <met0> [ <met1> ... ]");
66 
67  /* Read control parameters... */
68  read_ctl(argv[1], argc, argv, &ctl);
69  int dz = (int) scan_ctl(argv[1], argc, argv, "LAPSE_DZ", -1, "20", NULL);
70  double lat0 =
71  (int) scan_ctl(argv[1], argc, argv, "LAPSE_LAT0", -1, "-90", NULL);
72  double lat1 =
73  (int) scan_ctl(argv[1], argc, argv, "LAPSE_LAT1", -1, "90", NULL);
74  double z0 = (int) scan_ctl(argv[1], argc, argv, "LAPSE_Z0", -1, "0", NULL);
75  double z1 =
76  (int) scan_ctl(argv[1], argc, argv, "LAPSE_Z1", -1, "100", NULL);
77 
78  /* Loop over files... */
79  for (int i = 3; i < argc; i++) {
80 
81  /* Read meteorological data... */
82  if (!read_met(&ctl, argv[i], met))
83  continue;
84 
85  /* Get altitude and pressure profiles... */
86  for (int iz = 0; iz < met->np; iz++)
87  z[iz] = Z(met->p[iz]);
88  for (int iz = 0; iz <= 250; iz++) {
89  z2[iz] = 0.0 + 0.1 * iz;
90  p2[iz] = P(z2[iz]);
91  }
92 
93  /* Loop over grid points... */
94  for (int ix = 0; ix < met->nx; ix++)
95  for (int iy = 0; iy < met->ny; iy++) {
96 
97  /* Check latitude range... */
98  if (met->lat[iy] < lat0 || met->lat[iy] > lat1)
99  continue;
100 
101  /* Interpolate temperature profile... */
102  for (int iz = 0; iz < met->np; iz++)
103  t[iz] = met->t[ix][iy][iz];
104  spline(z, t, met->np, z2, t2, 251);
105 
106  /* Loop over vertical levels... */
107  for (int iz = 0; iz <= 250; iz++) {
108 
109  /* Check height range... */
110  if (z2[iz] < z0 || z2[iz] > z1)
111  continue;
112 
113  /* Check surface pressure... */
114  if (p2[iz] > met->ps[ix][iy])
115  continue;
116 
117  /* Get mean latitude and height... */
118  lat_mean += met->lat[iy];
119  z_mean += z2[iz];
120  np++;
121 
122  /* Get lapse rates within a vertical layer... */
123  int nlapse = 0;
124  double lapse_max = -1e99, lapse_min = 1e99, lapse_mean =
125  0, lapse_sig = 0;
126  for (int iz2 = iz + 1; iz2 <= iz + dz; iz2++) {
127  lapse_max =
128  GSL_MAX(LAPSE(p2[iz], t2[iz], p2[iz2], t2[iz2]), lapse_max);
129  lapse_min =
130  GSL_MIN(LAPSE(p2[iz], t2[iz], p2[iz2], t2[iz2]), lapse_min);
131  lapse_mean += LAPSE(p2[iz], t2[iz], p2[iz2], t2[iz2]);
132  lapse_sig += SQR(LAPSE(p2[iz], t2[iz], p2[iz2], t2[iz2]));
133  nlapse++;
134  }
135  lapse_mean /= nlapse;
136  lapse_sig = sqrt(GSL_MAX(lapse_sig / nlapse - SQR(lapse_mean), 0));
137 
138  /* Get histograms... */
139  int idx = (int) ((lapse_max - LAPSEMIN) / DLAPSE);
140  if (idx >= 0 && idx < IDXMAX) {
141  hist_max[idx]++;
142  nhist_max++;
143  }
144 
145  idx = (int) ((lapse_min - LAPSEMIN) / DLAPSE);
146  if (idx >= 0 && idx < IDXMAX) {
147  hist_min[idx]++;
148  nhist_min++;
149  }
150 
151  idx = (int) ((lapse_mean - LAPSEMIN) / DLAPSE);
152  if (idx >= 0 && idx < IDXMAX) {
153  hist_mean[idx]++;
154  nhist_mean++;
155  }
156 
157  idx = (int) ((lapse_sig - LAPSEMIN) / DLAPSE);
158  if (idx >= 0 && idx < IDXMAX) {
159  hist_sig[idx]++;
160  nhist_sig++;
161  }
162  }
163  }
164  }
165 
166  /* Create output file... */
167  printf("Write lapse rate data: %s\n", argv[2]);
168  if (!(out = fopen(argv[2], "w")))
169  ERRMSG("Cannot create file!");
170 
171  /* Write header... */
172  fprintf(out,
173  "# $1 = mean altitude [km]\n"
174  "# $2 = mean latitude [deg]\n"
175  "# $3 = lapse rate [K/km]\n"
176  "# $4 = counts of maxima per bin\n"
177  "# $5 = total number of maxima\n"
178  "# $6 = normalized frequency of maxima\n"
179  "# $7 = counts of minima per bin\n"
180  "# $8 = total number of minima\n"
181  "# $9 = normalized frequency of minima\n"
182  "# $10 = counts of means per bin\n"
183  "# $11 = total number of means\n"
184  "# $12 = normalized frequency of means\n"
185  "# $13 = counts of sigmas per bin\n"
186  "# $14 = total number of sigmas\n"
187  "# $15 = normalized frequency of sigmas\n\n");
188 
189  /* Write data... */
190  double nmax_max = 0, nmax_min = 0, nmax_mean = 0, nmax_sig = 0;
191  for (int idx = 0; idx < IDXMAX; idx++) {
192  nmax_max = GSL_MAX(hist_max[idx], nmax_max);
193  nmax_min = GSL_MAX(hist_min[idx], nmax_min);
194  nmax_mean = GSL_MAX(hist_mean[idx], nmax_mean);
195  nmax_sig = GSL_MAX(hist_sig[idx], nmax_sig);
196 
197  }
198  for (int idx = 0; idx < IDXMAX; idx++)
199  fprintf(out,
200  "%g %g %g %d %d %g %d %d %g %d %d %g %d %d %g\n",
201  z_mean / np, lat_mean / np, (idx + .5) * DLAPSE + LAPSEMIN,
202  hist_max[idx], nhist_max,
203  (double) hist_max[idx] / (double) nmax_max, hist_min[idx],
204  nhist_min, (double) hist_min[idx] / (double) nmax_min,
205  hist_mean[idx], nhist_mean,
206  (double) hist_mean[idx] / (double) nmax_mean, hist_sig[idx],
207  nhist_sig, (double) hist_sig[idx] / (double) nmax_sig);
208 
209  /* Close file... */
210  fclose(out);
211 
212  /* Free... */
213  free(met);
214 
215  return EXIT_SUCCESS;
216 }
