63 /* Read control parameters... */
64 mptrac_read_ctl(argv[1], argc, argv, &ctl);
65 const int geopot =
66 (int) scan_ctl(argv[1], argc, argv, "SAMPLE_GEOPOT", -1, "0", NULL);

73 const int grid_lat =

76 /* Read climatological data... */
77 mptrac_read_clim(&ctl, clim);
78
79 /* Read atmospheric data... */
80 if (!mptrac_read_atm(argv[3], &ctl, atm))

82
83 /* Create output file... */
84 LOG(1, "Write meteorological data file: %s", argv[2]);
85 if (!(out = fopen(argv[2], "w")))
86 ERRMSG("Cannot create file!");
87
88 /* Write header... */
89 MET_HEADER;
90
91 /* Loop over air parcels... */
92 for (int ip = 0; ip < atm->np; ip++) {
93
94 /* Get meteorological data... */
95 mptrac_get_met(&ctl, clim, atm->time[ip], &met0, &met1, dd);
96
97 /* Set reference pressure for interpolation... */
98 INTPOL_INIT;
99 double pref = atm->p[ip];
100 if (geopot) {
101 zref = Z(pref);
102 p0 = met0->p[0];
103 p1 = met0->p[met0->np - 1];
104 for (int it = 0; it < 24; it++) {
105 pref = 0.5 * (p0 + p1);
106 intpol_met_time_3d(met0, met0->z, met1, met1->z, atm->time[ip], pref,
107 atm->lon[ip], atm->lat[ip], &zm, ci, cw, 1);
108 if (zref > zm || !isfinite(zm))
109 p0 = pref;
110 else
111 p1 = pref;
112 }
113 pref = 0.5 * (p0 + p1);
114 }
115
116 /* Interpolate meteo data... */
117 INTPOL_TIME_ALL(atm->time[ip], pref, atm->lon[ip], atm->lat[ip]);
118
119 /* Make blank lines... */
120 if (ip == 0 || (grid_time && atm->time[ip] != time_old)
121 || (grid_z && atm->p[ip] != p_old)
122 || (grid_lon && atm->lon[ip] != lon_old)
123 || (grid_lat && atm->lat[ip] != lat_old))
124 fprintf(out, "\n");
125 time_old = atm->time[ip];
126 p_old = atm->p[ip];
127 lon_old = atm->lon[ip];
128 lat_old = atm->lat[ip];
129
130 /* Write data... */
131 fprintf(out,
132 "%.2f %g %g %g %g %g %g %g %g %g %g %g %g %g %g %g %g %g %g"
133 " %g %g %g %g %g %g %g %g %g %g %g %g %g %g %g %g %g %g %g"
134 " %g %g %g %g %g %g %g %g %g %g %g %g %g %g 1 1 1\n",
135 atm->time[ip], Z(atm->p[ip]), atm->lon[ip], atm->lat[ip],
136 atm->p[ip], t, u, v, w, h2o, o3, z, pv, ps, ts, zs, us, vs,
137 ess, nss, shf, lsm,
138 sst, pt, zt, tt, h2ot, lwc, rwc, iwc, swc, cc, cl, pct, pcb, plcl,
139 plfc, pel, cape, cin, RH(atm->p[ip], t, h2o), RHICE(atm->p[ip], t,
140 h2o),
141 TDEW(atm->p[ip], h2o), TICE(atm->p[ip], h2o),
142 nat_temperature(atm->p[ip], h2o,
143 clim_zm(&clim->hno3, atm->time[ip], atm->lat[ip],
144 atm->p[ip])), clim_zm(&clim->hno3,
145 atm->time[ip],
146 atm->lat[ip],
147 atm->p[ip]),
148 clim_oh(&ctl, clim, atm->time[ip], atm->lon[ip], atm->lat[ip],
149 atm->p[ip]), clim_zm(&clim->h2o2, atm->time[ip],
150 atm->lat[ip], atm->p[ip]),
151 clim_zm(&clim->ho2, atm->time[ip], atm->lat[ip], atm->p[ip]),
152 clim_zm(&clim->o1d, atm->time[ip], atm->lat[ip], atm->p[ip]), pbl,
153 o3c);
154 }
155
156 /* Close file... */
157 fclose(out);
158
159 /* Free... */

161 free(atm);
162 free(met0);
163 free(met1);
164 free(dd);
165
166 return EXIT_SUCCESS;
