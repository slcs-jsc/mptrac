39int main(
40 int argc,
41 char *argv[]) {
42
43 ctl_t ctl;
44
45 clim_t *clim;
46
47 atm_t *atm;
48
49 met_t *met0, *met1;
50
51 dd_t *dd;
52
53 /* Print usage information... */
54 USAGE;
55
56 /* Check arguments... */
57 if (argc < 3)
58 ERRMSG("Missing or invalid command-line arguments.\n\n"
59 "Usage: atm_init <ctl> <atm_out> [KEY VALUE ...]\n\n"
60 "Use -h for full help.");
61
62 /* Allocate... */
63 ALLOC(clim, clim_t, 1);
64 ALLOC(atm, atm_t, 1);
65 ALLOC(met0, met_t, 1);
66 ALLOC(met1, met_t, 1);
67 ALLOC(dd, dd_t, 1);
68
69 /* Read control parameters... */
70 mptrac_read_ctl(argv[1], argc, argv, &ctl);
71 const int ens =
72 (int) scan_ctl(argv[1], argc, argv, "INIT_ENS", -1, "0", NULL);
73 const double t0 = scan_ctl(argv[1], argc, argv, "INIT_T0", -1, "0", NULL);
74 const double t1 = scan_ctl(argv[1], argc, argv, "INIT_T1", -1, "0", NULL);
75 const double dt = scan_ctl(argv[1], argc, argv, "INIT_DT", -1, "1", NULL);
76 const double z0 = scan_ctl(argv[1], argc, argv, "INIT_Z0", -1, "0", NULL);
77 const double z1 = scan_ctl(argv[1], argc, argv, "INIT_Z1", -1, "0", NULL);
78 const double dz = scan_ctl(argv[1], argc, argv, "INIT_DZ", -1, "1", NULL);
79 const double lon0 =
80 scan_ctl(argv[1], argc, argv, "INIT_LON0", -1, "0", NULL);
81 const double lon1 =
82 scan_ctl(argv[1], argc, argv, "INIT_LON1", -1, "0", NULL);
83 const double dlon =
84 scan_ctl(argv[1], argc, argv, "INIT_DLON", -1, "1", NULL);
85 const double lat0 =
86 scan_ctl(argv[1], argc, argv, "INIT_LAT0", -1, "0", NULL);
87 const double lat1 =
88 scan_ctl(argv[1], argc, argv, "INIT_LAT1", -1, "0", NULL);
89 const double dlat =
90 scan_ctl(argv[1], argc, argv, "INIT_DLAT", -1, "1", NULL);
91 const double st = scan_ctl(argv[1], argc, argv, "INIT_ST", -1, "0", NULL);
92 const double sz = scan_ctl(argv[1], argc, argv, "INIT_SZ", -1, "0", NULL);
93 const double slon =
94 scan_ctl(argv[1], argc, argv, "INIT_SLON", -1, "0", NULL);
95 const double slat =
96 scan_ctl(argv[1], argc, argv, "INIT_SLAT", -1, "0", NULL);
97 const double sx = scan_ctl(argv[1], argc, argv, "INIT_SX", -1, "0", NULL);
98 const double ut = scan_ctl(argv[1], argc, argv, "INIT_UT", -1, "0", NULL);
99 const double uz = scan_ctl(argv[1], argc, argv, "INIT_UZ", -1, "0", NULL);
100 const double ulon =
101 scan_ctl(argv[1], argc, argv, "INIT_ULON", -1, "0", NULL);
102 const double ulat =
103 scan_ctl(argv[1], argc, argv, "INIT_ULAT", -1, "0", NULL);
104 const int even =
105 (int) scan_ctl(argv[1], argc, argv, "INIT_EVENLY", -1, "0", NULL);
106 const int col_mass =
107 (int) scan_ctl(argv[1], argc, argv, "INIT_COL_MASS", -1, "0", NULL);
108 const int well_mixed =
109 (int) scan_ctl(argv[1], argc, argv, "INIT_WELL_MIXED", -1, "0", NULL);
110 const int rep =
111 (int) scan_ctl(argv[1], argc, argv, "INIT_REP", -1, "1", NULL);
112 const int init_np =
113 (int) scan_ctl(argv[1], argc, argv, "INIT_NP", -1, "0", NULL);
114 const double m = scan_ctl(argv[1], argc, argv, "INIT_MASS", -1, "0", NULL);
115 const double vmr = scan_ctl(argv[1], argc, argv, "INIT_VMR", -1, "0", NULL);
116 const double bellrad =
117 scan_ctl(argv[1], argc, argv, "INIT_BELLRAD", -1, "0", NULL);
118 const int idx_offset =
119 (int) scan_ctl(argv[1], argc, argv, "INIT_IDX_OFFSET", -1, "0", NULL);
120
121 /* Check arguments... */
122 if (ctl.met_coord_type != 0 && even)
123 ERRMSG("INIT_EVENLY is only supported for lat/lon grids");
124 if (ctl.met_coord_type != 0 && bellrad > 0)
125 ERRMSG("INIT_BELLRAD is only supported for lat/lon grids");
126 if (rep <= 0)
127 ERRMSG("INIT_REP must be positive!");
128 if (init_np < 0 || init_np > NP)
129 ERRMSG("INIT_NP must be between 0 and NP!");
130 if (well_mixed && ctl.met_coord_type != 0)
131 ERRMSG("INIT_WELL_MIXED is only supported for lat/lon grids!");
132 if (well_mixed && !even)
133 ERRMSG("INIT_WELL_MIXED requires INIT_EVENLY=1!");
134 if (well_mixed && !col_mass)
135 ERRMSG("INIT_WELL_MIXED requires INIT_COL_MASS=1!");
136 if (init_np > 0 && !well_mixed)
137 ERRMSG("INIT_NP requires INIT_WELL_MIXED!");
138 if (col_mass && (t1 != t0 || st > 0 || ut > 0))
139 ERRMSG
140 ("INIT_COL_MASS/INIT_WELL_MIXED require INIT_T0 == INIT_T1 and no time scattering!");
141
142 /* Read climatological data... */
143 mptrac_read_clim(&ctl, clim);
144
145 /* Initialize random number generator... */
146 gsl_rng_env_setup();
147 gsl_rng *rng = gsl_rng_alloc(gsl_rng_default);
148
149 /* Get meteorological data for column-mass and well-mixed initialization... */
150 double ptop = 0, dpcolmax = 0;
151 if (col_mass) {
152 mptrac_get_met(&ctl, clim, t0, &met0, &met1, dd);
153 ptop = gsl_stats_min(met0->p, 1, (size_t) met0->np);
154 double psmax = -1;
155 for (int ix = 0; ix < met0->nx; ix++)
156 for (int iy = 0; iy < met0->ny; iy++) {
157 psmax = MAX(psmax, met0->ps[ix][iy]);
158 psmax = MAX(psmax, met1->ps[ix][iy]);
159 }
160 dpcolmax = psmax - ptop;
161 }
162
163 /* Create grid... */
164 int ntry = 0;
165 const int ntrymax = init_np > 0 ? 1000000 * init_np : 0;
166 do {
167 for (double t = t0; t <= t1 && !(init_np > 0 && atm->np >= init_np);
168 t += dt)
169 for (double z = z0; z <= z1 && !(init_np > 0 && atm->np >= init_np);
170 z += dz)
171 for (double lon = lon0;
172 lon <= lon1 && !(init_np > 0 && atm->np >= init_np); lon += dlon)
173 for (double lat = lat0;
174 lat <= lat1 && !(init_np > 0 && atm->np >= init_np);
175 lat += dlat)
176 for (int irep = 0;
177 irep < rep && !(init_np > 0 && atm->np >= init_np); irep++) {
178
179 /* Check number of attempts... */
180 if (init_np > 0 && ++ntry > ntrymax)
181 ERRMSG
182 ("INIT_NP could not be reached by INIT_WELL_MIXED sampling!");
183
184 /* Set time... */
185 double rg = gsl_ran_gaussian_ziggurat(rng, st / 2.3548);
186 double ru = ut * (gsl_rng_uniform(rng) - 0.5);
187 atm->time[atm->np] = (t + rg + ru);
188
189 /* Set pressure... */
190 rg = gsl_ran_gaussian_ziggurat(rng, sz / 2.3548);
191 ru = uz * (gsl_rng_uniform(rng) - 0.5);
192 atm->p[atm->np] = P(z + rg + ru);
193
194 /* Set horizontal position... */
195 rg = gsl_ran_gaussian_ziggurat(rng, slon / 2.3548);
196 const double sx_coord =
197 ctl.met_coord_type == 0 ? DX2DEG(sx, lat) : sx;
198 double rx = gsl_ran_gaussian_ziggurat(rng, sx_coord / 2.3548);
199 ru = ulon * (gsl_rng_uniform(rng) - 0.5);
200 atm->lon[atm->np] = (lon + rg + rx + ru);
201
202 /* Set ensemble index... */
203 if (ctl.qnt_ens >= 0)
204 atm->q[ctl.qnt_ens][atm->np] = ens;
205
206 /* Apply cosine-latitude weighting... */
207 do {
208 rg = gsl_ran_gaussian_ziggurat(rng, slat / 2.3548);
209 const double sy_coord =
210 ctl.met_coord_type == 0 ? DY2DEG(sx) : sx;
211 rx = gsl_ran_gaussian_ziggurat(rng, sy_coord / 2.3548);
212 ru = ulat * (gsl_rng_uniform(rng) - 0.5);
213 atm->lat[atm->np] = (lat + rg + rx + ru);
214 } while (even && gsl_rng_uniform(rng) >
215 fabs(cos(DEG2RAD(atm->lat[atm->np]))));
216
217 /* Apply cosine bell (Williamson et al., 1992)... */
218 if (bellrad > 0) {
219 double x0[3], x1[3];
220 geo2cart(0.0, 0.5 * (lon0 + lon1), 0.5 * (lat0 + lat1), x0);
221 geo2cart(0.0, atm->lon[atm->np], atm->lat[atm->np], x1);
222 const double rad =
223 RE * acos(DOTP(x0, x1) / sqrt(DOTP(x0, x0)) /
224 sqrt(DOTP(x1, x1)));
225 if (rad > bellrad)
226 continue;
227 if (ctl.qnt_m >= 0)
228 atm->q[ctl.qnt_m][atm->np] =
229 0.5 * (1. + cos(M_PI * rad / bellrad));
230 if (ctl.qnt_vmr >= 0)
231 atm->q[ctl.qnt_vmr][atm->np] =
232 0.5 * (1. + cos(M_PI * rad / bellrad));
233 }
234
235 /* Apply column-mass and globally well-mixed air-mass weighting... */
236 if (col_mass) {
237 INTPOL_INIT;
238 double ps;
239 intpol_met_time_2d(met0, met0->ps, met1, met1->ps,
240 atm->time[atm->np], atm->lon[atm->np],
241 atm->lat[atm->np], &ps, ci, cw, 1);
242 const double dpcol = ps - ptop;
243 if (gsl_rng_uniform(rng) > dpcol / dpcolmax)
244 continue;
245 if (well_mixed)
246 atm->p[atm->np] = ptop + dpcol * gsl_rng_uniform(rng);
247 }
248
249 /* Set particle counter... */
250 if ((++atm->np) > NP)
251 ERRMSG("Too many particles!");
252 }
253
254 } while (init_np > 0 && atm->np < init_np);
255
256 /* Check number of air parcels... */
257 if (atm->np <= 0)
258 ERRMSG("Did not create any air parcels!");
259
260 /* Initialize mass... */
261 if (ctl.qnt_m >= 0 && bellrad <= 0)
262 for (int ip = 0; ip < atm->np; ip++)
263 atm->q[ctl.qnt_m][ip] = m / atm->np;
264
265 /* Initialize volume mixing ratio... */
266 if (ctl.qnt_vmr >= 0 && bellrad <= 0)
267 for (int ip = 0; ip < atm->np; ip++)
268 atm->q[ctl.qnt_vmr][ip] = vmr;
269
270 /* Initialize air parcel index... */
271 if (ctl.qnt_idx >= 0)
272 for (int ip = 0; ip < atm->np; ip++)
273 atm->q[ctl.qnt_idx][ip] = idx_offset + ip;
274
275 /* Initialize age of air... */
276 if (ctl.qnt_aoa >= 0)
277 for (int ip = 0; ip < atm->np; ip++)
278 atm->q[ctl.qnt_aoa][ip] = atm->time[ip];
279
280 /* Save data... */
281 mptrac_write_atm(argv[2], &ctl, atm, 0);
282
283 /* Free... */
284 gsl_rng_free(rng);
285 free(clim);
286 free(atm);
287 free(met0);
288 free(met1);
289 free(dd);
290
291 return EXIT_SUCCESS;
292}
