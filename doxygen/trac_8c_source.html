31int main(
32 int argc,
33 char *argv[]) {
34
35 ctl_t ctl;
36
37 atm_t *atm;
38
39 cache_t *cache;
40
41 clim_t *clim;
42
43 met_t *met0, *met1;
44
45#ifdef ASYNCIO
46 met_t *met0TMP, *met1TMP, *mets;
47 ctl_t ctlTMP;
48#endif
49
50 FILE *dirlist;
51
52 char dirname[LEN], filename[2 * LEN];
53
54 double *dt, *rs;
55
56 int ntask = -1, rank = 0, size = 1;
57
58 /* Start timers... */
59 START_TIMERS;
60
61 /* Initialize MPI... */
62#ifdef MPI
63 SELECT_TIMER("MPI_INIT", "INIT", NVTX_CPU);
64 MPI_Init(&argc, &argv);
65 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
66 MPI_Comm_size(MPI_COMM_WORLD, &size);
67#endif
68
69 /* Initialize GPUs... */
70#ifdef _OPENACC
71 SELECT_TIMER("ACC_INIT", "INIT", NVTX_GPU);
72 if (acc_get_num_devices(acc_device_nvidia) <= 0)
73 ERRMSG("Not running on a GPU device!");
74 acc_set_device_num(rank % acc_get_num_devices(acc_device_nvidia),
75 acc_device_nvidia);
76 acc_device_t device_type = acc_get_device_type();
77 acc_init(device_type);
78#endif
79
80 /* Check arguments... */
81 if (argc < 4)
82 ERRMSG("Give parameters: <dirlist> <ctl> <atm_in>");
83
84 /* Open directory list... */
85 if (!(dirlist = fopen(argv[1], "r")))
86 ERRMSG("Cannot open directory list!");
87
88 /* Loop over directories... */
89 while (fscanf(dirlist, "%4999s", dirname) != EOF) {
90
91 /* MPI parallelization... */
92 if ((++ntask) % size != rank)
93 continue;
94
95 /* ------------------------------------------------------------
96 Initialize model run...
97 ------------------------------------------------------------ */
98
99 /* Allocate... */
100 SELECT_TIMER("ALLOC", "MEMORY", NVTX_CPU);
101 ALLOC(atm, atm_t, 1);
102 ALLOC(cache, cache_t, 1);
103 ALLOC(clim, clim_t, 1);
104 ALLOC(met0, met_t, 1);
105 ALLOC(met1, met_t, 1);
106#ifdef ASYNCIO
107 ALLOC(met0TMP, met_t, 1);
108 ALLOC(met1TMP, met_t, 1);
109#endif
110 ALLOC(dt, double,
111 NP);
112 ALLOC(rs, double,
113 3 * NP + 1);
114
115 /* Create data region on GPUs... */
116#ifdef _OPENACC
117 SELECT_TIMER("CREATE_DATA_REGION", "MEMORY", NVTX_GPU);
118#ifdef ASYNCIO
119#pragma acc enter data create(atm[:1], cache[:1], clim[:1], ctl, ctlTMP, met0[:1], met1[:1], met0TMP[:1], met1TMP[:1], dt[:NP], rs[:3 * NP])
120#else
121#pragma acc enter data create(atm[:1], cache[:1], clim[:1], ctl, met0[:1], met1[:1], dt[:NP], rs[:3 * NP])
122#endif
123#endif
124
125 /* Read control parameters... */
126 sprintf(filename, "%s/%s", dirname, argv[2]);
127 read_ctl(filename, argc, argv, &ctl);
128
129 /* Read climatological data... */
130 read_clim(&ctl, clim);
131
132 /* Read atmospheric data... */
133 sprintf(filename, "%s/%s", dirname, argv[3]);
134 if (!read_atm(filename, &ctl, atm))
135 ERRMSG("Cannot open file!");
136
137 /* Initialize timesteps... */
138 module_timesteps_init(&ctl, atm);
139
140 /* Initialize random number generator... */
141 module_rng_init(ntask);
142
143 /* Initialize meteo data... */
144#ifdef ASYNCIO
145 ctlTMP = ctl;
146#endif
147 get_met(&ctl, clim, ctl.t_start, &met0, &met1);
148 if (ctl.dt_mod > fabs(met0->lon[1] - met0->lon[0]) * 111132. / 150.)
149 WARN("Violation of CFL criterion! Check DT_MOD!");
150#ifdef ASYNCIO
151 get_met(&ctlTMP, clim, ctlTMP.t_start, &met0TMP, &met1TMP);
152#endif
153
154 /* Initialize isosurface data... */
155 if (ctl.isosurf >= 1 && ctl.isosurf <= 4)
156 module_isosurf_init(&ctl, met0, met1, atm, cache);
157
158 /* Initialize advection... */
159 module_advect_init(&ctl, met0, met1, atm);
160
161 /* Initialize chemistry... */
162 module_chem_init(&ctl, clim, met0, met1, atm);
163
164 /* Update GPU... */
165#ifdef _OPENACC
166 SELECT_TIMER("UPDATE_DEVICE", "MEMORY", NVTX_H2D);
167#pragma acc update device(atm[:1], cache[:1], clim[:1], ctl)
168#endif
169
170 /* ------------------------------------------------------------
171 Loop over timesteps...
172 ------------------------------------------------------------ */
173
174 /* Loop over timesteps... */
175#ifdef ASYNCIO
176 omp_set_nested(1);
177 int ompTrdnum = omp_get_max_threads();
178#endif
179 for (double t = ctl.t_start;
180 ctl.direction * (t - ctl.t_stop) < ctl.dt_mod;
181 t += ctl.direction * ctl.dt_mod) {
182#ifdef ASYNCIO
183#pragma omp parallel num_threads(2)
184 {
185#endif
186
187 /* Adjust length of final time step... */
188 if (ctl.direction * (t - ctl.t_stop) > 0)
189 t = ctl.t_stop;
190
191 /* Set time steps of air parcels... */
192 module_timesteps(&ctl, met0, atm, dt, t);
193
194 /* Get meteo data... */
195#ifdef ASYNCIO
196#pragma acc wait(5)
197#pragma omp barrier
198 if (omp_get_thread_num() == 0) {
199
200 /* Pointer swap... */
201 if (t != ctl.t_start) {
202 mets = met0;
203 met0 = met0TMP;
204 met0TMP = mets;
205
206 mets = met1;
207 met1 = met1TMP;
208 met1TMP = mets;
209 }
210#endif
211#ifndef ASYNCIO
212 if (t != ctl.t_start)
213 get_met(&ctl, clim, t, &met0, &met1);
214#endif
215
216 /* Sort particles... */
217 if (ctl.sort_dt > 0 && fmod(t, ctl.sort_dt) == 0)
218 module_sort(&ctl, met0, atm);
219
220 /* Check positions (initial)... */
221 module_position(&ctl, met0, met1, atm, dt);
222
223 /* Advection... */
224 if (ctl.advect > 0)
225 module_advect(&ctl, met0, met1, atm, dt);
226
227 /* Turbulent diffusion... */
228 if (ctl.turb_dx_trop > 0 || ctl.turb_dz_trop > 0
229 || ctl.turb_dx_strat > 0 || ctl.turb_dz_strat > 0)
230 module_diffusion_turb(&ctl, clim, atm, dt, rs);
231
232 /* Mesoscale diffusion... */
233 if (ctl.turb_mesox > 0 || ctl.turb_mesoz > 0)
234 module_diffusion_meso(&ctl, met0, met1, atm, cache, dt, rs);
235
236 /* Convection... */
237 if (ctl.conv_cape >= 0
238 && (ctl.conv_dt <= 0 || fmod(t, ctl.conv_dt) == 0))
239 module_convection(&ctl, met0, met1, atm, dt, rs);
240
241 /* Sedimentation... */
242 if (ctl.qnt_rp >= 0 && ctl.qnt_rhop >= 0)
243 module_sedi(&ctl, met0, met1, atm, dt);
244
245 /* Isosurface... */
246 if (ctl.isosurf >= 1 && ctl.isosurf <= 4)
247 module_isosurf(&ctl, met0, met1, atm, cache, dt);
248
249 /* Check positions (final)... */
250 module_position(&ctl, met0, met1, atm, dt);
251
252 /* Interpolate meteo data... */
253 if (ctl.met_dt_out > 0
254 && (ctl.met_dt_out < ctl.dt_mod
255 || fmod(t, ctl.met_dt_out) == 0))
256 module_meteo(&ctl, clim, met0, met1, atm, dt);
257
258 /* Check boundary conditions (initial)... */
259 if ((ctl.bound_lat0 < ctl.bound_lat1)
260 && (ctl.bound_p0 > ctl.bound_p1))
261 module_bound_cond(&ctl, clim, met0, met1, atm, dt);
262
263 /* Initialize quantity of total loss rate... */
264 if (ctl.qnt_loss_rate >= 0) {
265 PARTICLE_LOOP(0, atm->np, 1, "acc data present(ctl, atm)") {
266 atm->q[ctl.qnt_loss_rate][ip] = 0;
267 }
268 }
269
270 /* Decay of particle mass... */
271 if (ctl.tdec_trop > 0 && ctl.tdec_strat > 0)
272 module_decay(&ctl, clim, atm, dt);
273
274 /* Interparcel mixing... */
275 if (ctl.mixing_trop >= 0 && ctl.mixing_strat >= 0
276 && (ctl.mixing_dt <= 0 || fmod(t, ctl.mixing_dt) == 0))
277 module_mixing(&ctl, clim, atm, t);
278
279 /* Calculate the tracer vmr in the chemistry grid... */
280 if (ctl.qnt_Cx > 0
281 && (ctl.oh_chem_reaction != 0 || ctl.h2o2_chem_reaction != 0
282 || (ctl.kpp_chem && fmod(t, ctl.dt_kpp) == 0)))
283 module_chemgrid(&ctl, met0, met1, atm, t);
284
285 /* OH chemistry... */
286 if (ctl.oh_chem_reaction != 0)
287 module_oh_chem(&ctl, clim, met0, met1, atm, dt);
288
289 /* H2O2 chemistry (for SO2 aqueous phase oxidation)... */
290 if (ctl.h2o2_chem_reaction != 0)
291 module_h2o2_chem(&ctl, clim, met0, met1, atm, dt);
292
293 /* First-order tracer chemistry... */
294 if (ctl.tracer_chem)
295 module_tracer_chem(&ctl, clim, met0, met1, atm, dt);
296
297 /* KPP chemistry... */
298 if (ctl.kpp_chem && fmod(t, ctl.dt_kpp) == 0) {
299#ifdef KPP
300 module_kpp_chem(&ctl, clim, met0, met1, atm, dt);
301#else
302 ERRMSG("Code was compiled without KPP!");
303#endif
304 }
305
306 /* Wet deposition... */
307 if ((ctl.wet_depo_ic_a > 0 || ctl.wet_depo_ic_h[0] > 0)
308 && (ctl.wet_depo_bc_a > 0 || ctl.wet_depo_bc_h[0] > 0))
309 module_wet_deposition(&ctl, met0, met1, atm, dt);
310
311 /* Dry deposition... */
312 if (ctl.dry_depo_vdep > 0)
313 module_dry_deposition(&ctl, met0, met1, atm, dt);
314
315 /* Check boundary conditions (final)... */
316 if ((ctl.bound_lat0 < ctl.bound_lat1)
317 && (ctl.bound_p0 > ctl.bound_p1))
318 module_bound_cond(&ctl, clim, met0, met1, atm, dt);
319
320 /* Write output... */
321 write_output(dirname, &ctl, met0, met1, atm, t);
322#ifdef ASYNCIO
323 } else {
324 omp_set_num_threads(ompTrdnum);
325 if (ctl.direction * (t - ctl.t_stop + ctl.direction * ctl.dt_mod) <
326 ctl.dt_mod)
327 get_met(&ctl, clim, t + (ctl.direction * ctl.dt_mod), &met0TMP,
328 &met1TMP);
329 }
330 }
331#endif
332 }
333
334#ifdef ASYNCIO
335 omp_set_num_threads(ompTrdnum);
336#endif
337
338 /* ------------------------------------------------------------
339 Finalize model run...
340 ------------------------------------------------------------ */
341
342 /* Flush output buffer... */
343 fflush(NULL);
344
345 /* Report problem size... */
346 LOG(1, "SIZE_NP = %d", atm->np);
347 LOG(1, "SIZE_MPI_TASKS = %d", size);
348 LOG(1, "SIZE_OMP_THREADS = %d", omp_get_max_threads());
349#ifdef _OPENACC
350 LOG(1, "SIZE_ACC_DEVICES = %d", acc_get_num_devices(acc_device_nvidia));
351#else
352 LOG(1, "SIZE_ACC_DEVICES = %d", 0);
353#endif
354
355 /* Report memory usage... */
356 LOG(1, "MEMORY_ATM = %g MByte", sizeof(atm_t) / 1024. / 1024.);
357 LOG(1, "MEMORY_CACHE = %g MByte", sizeof(cache_t) / 1024. / 1024.);
358 LOG(1, "MEMORY_CLIM = %g MByte", sizeof(clim_t) / 1024. / 1024.);
359 LOG(1, "MEMORY_METEO = %g MByte", 2 * sizeof(met_t) / 1024. / 1024.);
360 LOG(1, "MEMORY_DYNAMIC = %g MByte", (3 * NP * sizeof(int)
361 + 4 * NP * sizeof(double)
362 + EX * EY * EP * sizeof(float)) /
363 1024. / 1024.);
364 LOG(1, "MEMORY_STATIC = %g MByte", (EX * EY * EP * sizeof(float)) /
365 1024. / 1024.);
366
367 /* Delete data region on GPUs... */
368#ifdef _OPENACC
369 SELECT_TIMER("DELETE_DATA_REGION", "MEMORY", NVTX_GPU);
370#ifdef ASYNCIO
371#pragma acc exit data delete (ctl, atm, cache, clim, met0, met1, dt, rs, met0TMP, met1TMP)
372#else
373#pragma acc exit data delete (ctl, atm, cache, clim, met0, met1, dt, rs)
374#endif
375#endif
376
377 /* Free... */
378 SELECT_TIMER("FREE", "MEMORY", NVTX_CPU);
379 free(atm);
380 free(cache);
381 free(clim);
382 free(met0);
383 free(met1);
384#ifdef ASYNCIO
385 free(met0TMP);
386 free(met1TMP);
387#endif
388 free(dt);
389 free(rs);
390
391 /* Report timers... */
392 PRINT_TIMERS;
393 }
394
395 /* Finalize MPI... */
396#ifdef MPI
397 MPI_Finalize();
398#endif
399
400 /* Stop timers... */
401 STOP_TIMERS;
402
403 return EXIT_SUCCESS;
404}
